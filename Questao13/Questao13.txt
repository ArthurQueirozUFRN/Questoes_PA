#include <stdio.h>
#include <stdlib.h>

void inserir(int tam, float num[]){
  printf("Digite os numeros: \n");
  for(int i=0; i<tam; i++){
    scanf("%f", &num[i]);
  }
}

void ordenar(int tam, float num[]){
  for( int i=0; i<tam; i++ ){
    for( int j=i+1; j<tam; j++ ){
      if( num[i] > num[j] ){
        float aux;
        aux = num[i];
        num[i] = num[j];
        num[j] = aux;
      }
    }
  }
}

int main() {
  int tam;
  float *p;

  printf("Quantos numeros? \n");
  scanf("%d", &tam);

  p = malloc(tam * sizeof(float));

  inserir(tam, p);

  ordenar(tam, p);

  for(int i=0; i<tam; i++){
    printf("%.2f ", p[i]);
  }

  return 0;
}
